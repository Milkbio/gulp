第一步，安装所需的包，用npm安装，如下：

npm install gulp gulp-rev run-sequence gulp-rev-collector --save-dev

第二步，在gulpfile.js文件中写上配置：

  1. //引入gulp和gulp插件
  2. var gulp = require('gulp'),
  3.     runSequence = require('run-sequence'),
  4.     rev = require('gulp-rev'),
  5.     revCollector = require('gulp-rev-collector');
  6.  
  7. //定义css、js源文件路径
  8. var cssSrc = './app/static/css/*.css',
  9.     jsSrc = './app/static/js/admin/*.js';
  10.  
  11.  
  12. //CSS生成文件hash编码并生成 rev-manifest.json文件名对照映射
  13. gulp.task('revCss', function(){
  14.     return gulp.src(cssSrc)
  15.         .pipe(rev())
  16.         .pipe(gulp.dest('./rev/css'))
  17.         .pipe(rev.manifest())
  18.         .pipe(gulp.dest('./rev/css'));
  19. });
  20.  
  21.  
  22. //js生成文件hash编码并生成 rev-manifest.json文件名对照映射
  23. gulp.task('revJs', function(){
  24.     return gulp.src(jsSrc)
  25.         .pipe(rev())                                //给文件添加hash编码
  26.         .pipe(gulp.dest('./rev/js/admin'))
  27.         .pipe(rev.manifest())                       //生成rev-mainfest.json文件作为记录
  28.         .pipe(gulp.dest('./rev/js'));
  29. });
  30.  
  31.  
  32. //Html替换css、js文件版本
  33. gulp.task('revHtmlCss', function () {
  34.     return gulp.src(['./rev/css/*.json', './app/index.html'])
  35.         .pipe(revCollector())                         //替换html中对应的记录
  36.         .pipe(gulp.dest('./View'));                     //输出到该文件夹中
  37. });
  38. gulp.task('revHtmlJs', function () {
  39.     return gulp.src(['./rev/js/*.json', './View/index.html'])
  40.         .pipe(revCollector())
  41.         .pipe(gulp.dest('./View'));
  42. });
  43.  
  44.  
  45. //开发构建
  46. gulp.task('dev', function (done) {
  47.     condition = false;
  48.     //依次顺序执行
  49.     runSequence(
  50.         ['revCss'],
  51.         ['revHtmlCss'],
  52.         ['revJs'],
  53.         ['revHtmlJs'],
  54.         done);
  55. });
第三步，打开cmd，切换到项目所在的目录，运行 

gulp dev

ok,完工！！！
